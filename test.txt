            // Create skeleton loader
            const skeleton = document.createElement('div');
            skeleton.className = 'loading-skeleton w-full h-full rounded-lg';

            const pageImages = images.slice(startIndex, endIndex);
            
            // Hide load more button if we're at the end
            const loadMoreButton = document.getElementById('load-more');
            if (endIndex >= images.length) {
                loadMoreButton.classList.add('hidden');
            } else {
                loadMoreButton.classList.remove('hidden');
            }
            
            // Split images between containers
            const halfwayPoint = Math.ceil(pageImages.length / 2);
            
            // Add images to first container
            pageImages.slice(0, halfwayPoint).forEach(imageData => {
                container1.appendChild(createImageElement(imageData));
            });
            
            // Add images to second container
            pageImages.slice(halfwayPoint).forEach(imageData => {
                container2.appendChild(createImageElement(imageData));
            });
            
            // Setup click handlers for modal
            setupImageModal();
        }

function setupFilters() {
    const filterButtons = document.querySelectorAll('.filter-button');
    
    filterButtons.forEach(button => {
        button.addEventListener('click', function() {
            // Remove active class from all buttons
            filterButtons.forEach(btn => btn.classList.remove('active', 'bg-custom', 'text-white'));
            filterButtons.forEach(btn => btn.classList.add('bg-gray-200', 'text-gray-700'));
            
            // Add active class to clicked button
            this.classList.add('active', 'bg-custom', 'text-white');
            this.classList.remove('bg-gray-200', 'text-gray-700');
            
            const filter = this.getAttribute('data-filter');
            
            // Filter images
            if (filter === 'all') {
                filteredImages = allImageData;
            } else {
                filteredImages = allImageData.filter(img => img.category === filter);
            }
            
            // Reset pagination and display filtered images
            currentPage = 1;
            displayImages(filteredImages, currentPage);
        });
    });
}

// Load more functionality
function setupLoadMore() {
    const loadMoreButton = document.getElementById('load-more');
    
    loadMoreButton.addEventListener('click', function() {
        currentPage++;
        displayImages(filteredImages, currentPage);
    });
}

// Setup image modal functionality
function setupImageModal() {
    const imageContainers = document.querySelectorAll('.image-container');
    const modal = document.getElementById('image-modal');
    const modalImage = document.getElementById('modal-image');
    const modalTitle = document.getElementById('modal-title');
    const modalDescription = document.getElementById('modal-description');
    const modalClose = document.getElementById('modal-close');
    const modalPrev = document.getElementById('modal-prev');
    const modalNext = document.getElementById('modal-next');
    
    let currentImageIndex = 0;
    
    // Open modal when clicking on an image
    imageContainers.forEach((container, index) => {
        container.addEventListener('click', function() {
            const category = this.getAttribute('data-category');
            const visibleImages = Array.from(document.querySelectorAll('.image-container')).filter(
                cont => !cont.style.display || cont.style.display !== 'none'
            );
            
            currentImageIndex = visibleImages.indexOf(this);
            const imageArray = filteredImages;
            
            if (currentImageIndex !== -1 && currentImageIndex < imageArray.length) {
                const imageData = imageArray[currentImageIndex];
                modalImage.src = imageData.src;
                modalTitle.textContent = imageData.title;
                modalDescription.textContent = imageData.description;
                modal.classList.add('active');
                
                // Update prev/next button visibility
                updateNavigationButtons(currentImageIndex, visibleImages.length);
            }
        });
    });
    
    // Close modal
    modalClose.addEventListener('click', function() {
        modal.classList.remove('active');
    });
    
    // Navigate to previous image
    modalPrev.addEventListener('click', function() {
        if (currentImageIndex > 0) {
            currentImageIndex--;
            updateModalImage(currentImageIndex);
            updateNavigationButtons(currentImageIndex, filteredImages.length);
        }
    });
    
    // Navigate to next image
    modalNext.addEventListener('click', function() {
        if (currentImageIndex < filteredImages.length - 1) {
            currentImageIndex++;
            updateModalImage(currentImageIndex);
            updateNavigationButtons(currentImageIndex, filteredImages.length);
        }
    });
    
    // Close modal when clicking outside of content
    modal.addEventListener('click', function(e) {
        if (e.target === modal) {
            modal.classList.remove('active');
        }
    });
    
    // Handle keyboard navigation
    document.addEventListener('keydown', function(e) {
        if (!modal.classList.contains('active')) return;
        
        if (e.key === 'Escape') {
            modal.classList.remove('active');
        } else if (e.key === 'ArrowLeft') {
            modalPrev.click();
        } else if (e.key === 'ArrowRight') {
            modalNext.click();
        }
    });
}

// Update modal image
function updateModalImage(index) {
    const modalImage = document.getElementById('modal-image');
    const modalTitle = document.getElementById('modal-title');
    const modalDescription = document.getElementById('modal-description');
    
    const imageData = filteredImages[index];
    modalImage.src = imageData.src;
    modalTitle.textContent = imageData.title;
    modalDescription.textContent = imageData.description;
}

// Update navigation buttons visibility
function updateNavigationButtons(index, total) {
    const modalPrev = document.getElementById('modal-prev');
    const modalNext = document.getElementById('modal-next');
    
    modalPrev.style.visibility = index === 0 ? 'hidden' : 'visible';
    modalNext.style.visibility = index === total - 1 ? 'hidden' : 'visible';
}
